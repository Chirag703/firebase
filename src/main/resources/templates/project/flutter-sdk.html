Flutter SDK Setup

Integrate your Flutter app with Firebase Console using our SDK.

1 Add Dependency

Add the Firebase Console SDK to your pubspec.yaml:

dependencies:
  flutter:
    sdk: flutter
  firebase_console_sdk:
    path: ./firebase_console_sdk  # or from pub.dev
2 Initialize SDK

Initialize Firebase in your main.dart:

import 'package:firebase_console_sdk/firebase_console_sdk.dart';

void main() async {
  WidgetsFlutterBinding.ensureInitialized();

  await FirebaseApp.initialize(
    apiKey: 'YOUR_API_KEY',
    projectId: 'YOUR_PROJECT_ID',
    baseUrl: 'http://localhost:8080',
  );

  runApp(MyApp());
}
3 Authentication

Sign Up

final auth = FirebaseAuth.instance;

final user = await auth.signUp(
  email: 'user@example.com',
  password: 'password123',
  name: 'John Doe',
);

Sign In

final user = await auth.signIn(
  email: 'user@example.com',
  password: 'password123',
);

print(user.uid);
print(user.email);
4 Database Access 
final db = FirebaseDatabase.instance;

// Read data
final data = await db.get('users', 'user_123');

// Write data
await db.set('users', 'user_123', {
  'name': 'John Doe',
  'email': 'john@example.com',
});

// Push new document (auto-ID)
final id = await db.push('messages', {
  'text': 'Hello World',
  'timestamp': DateTime.now().toIso8601String(),
});

// Delete
await db.delete('users', 'user_123');
